Quick Start
# Install PearlPool
curl -sL https://raw.githubusercontent.com/EasyPoolPearl/pearlpool/main/install.sh | bash
# Start the pool
./pearlpool --wallet prl1pYOUR_WALLET_ADDRESS
# Connect your miner
alpha-miner --pool stratum+tcp://YOUR_HOST:3333 --wallet prl1pYOUR_ADDR
Stratum stratum+tcp://0.0.0.0:3333
